	}

		NOT = { 
			stability = 1
		}
		NOT = { 
			has_country_flag = liberal_constitution
		}
	}
	can_stop = { 
		OR = { 
			has_any_disaster = yes
			stability = 2
		}
	}
	progress = { 
		modifier = { 
			factor = 0.5
			government = monarchy
			absolutism = 70
		}
		modifier = { 
			factor = 0.5
			AND = { 
				government = monarchy
				OR = { 
					has_reform = Pol_Assembly2
					has_reform = Pol_Assembly3
				}
				OR = { 
					has_reform = Pol_Structure2
					has_reform = Pol_Structure3
				}

					has_reform = Pol_Centralisation3

				}

			factor = 0.5

				OR = { 
					has_reform = Pol_Nobility3

				}
				has_reform = Pol_Structure3
			}
		}
		modifier = { 
			factor = 0.5
			AND = { 
				OR = { 
					AND = { 
						government = republic

				}
			}
		}
		modifier = { 
			factor = 1

			factor = 0.5

				absolutism = 50

		}

			factor = 1.0

				absolutism = 75
			}

			factor = 2.0

		}

			factor = 1

		}
	}
	can_end = { 
		OR = { 
			current_size_of_parliament = 1
			AND = { 
				stability = 1

					any_owned_province = { 
						has_province_modifier = revolutionaries_organizing
					}
				}
				NOT = { 
					num_of_revolts = 1
				}
				NOT = { 
					num_of_rebel_controlled_provinces = 1
				}
			}

	modifier = { 
		global_unrest = 10
		legitimacy = -3
		reduced_liberty_desire = -20
	}
	on_start = aspiration_for_liberty.1
	on_end = aspiration_for_liberty.4

		events = { }

			1000 = 0

		}
	}
}

french_revolution = { 
	potential = { 
		NOT = { 
			has_country_flag = had_revolution
		}
		tag = FRA
		AND = { 
			OR = { 
				is_subject = no
				is_subject_of_type = tributary_state
				is_subject_of_type = close_tributary_state
			}
		}
		uses_doom = no
		revolution_target_exists = no
		government = monarchy
		NOT = { 
			AND = { 
				government = monarchy
				culture_group = chinese_group
				has_reform = Pol_Succession5
			}
		}
	}
	can_start = { 
		current_age = age_of_revolutions
		num_of_cities = 30
		has_any_disaster = no
		OR = { 
			NOT = { 
				stability = 0
			}
			num_of_loans = 50
			inflation = 15
			war_exhaustion = 10
			NOT = { 
				prestige = 0
			}
		}
	}
	can_stop = { 
		OR = { 
			has_any_disaster = yes
			stability = 3
		}
	}
	progress = { 
		modifier = { 
			factor = 1
			NOT = { 
				prestige = 0
			}
		}
		modifier = { 
			factor = 1
			war_exhaustion = 5
		}
		modifier = { 
			factor = 0.5
			num_of_loans = 10
			NOT = { 
				num_of_loans = 25
			}
		}
		modifier = { 
			factor = 0.5
			absolutism = 70
		}
		modifier = { 
			factor = 1
			absolutism = 80
		}
		modifier = { 
			factor = 1
			num_of_loans = 25
		}
		modifier = { 
			factor = 2
			is_bankrupt = yes
		}
	}
	can_end = { 
		OR = { 
			AND = { 
				stability = 3
				NOT = { 
					num_of_rebel_controlled_provinces = 1
				}
				government = monarchy
			}
			AND = { 
				AND = { 
					government = republic
					is_revolutionary = yes
				}
				stability = 3
				NOT = { 
					num_of_rebel_controlled_provinces = 1
				}
			}
			AND = { 
				AND = { 
					government = republic
					is_revolutionary = yes
				}
				is_revolution_target = yes
				NOT = { 
					num_of_rebel_controlled_provinces = 1
				}
				has_ruler_flag = bonaparte_flag
			}
		}
	}
	modifier = { 
		global_unrest = 10
		stability_cost_modifier = 0.1
	}
	on_start = french_revolution.1
	on_end = french_revolution.18
	on_monthly = { 
		events = { french_revolution.2}
		random_events = { 
			1000 = 0
			100 = french_revolution.3
			100 = french_revolution.4
			100 = french_revolution.5
			100 = french_revolution.7
			100 = french_revolution.8
			100 = french_revolution.10
			100 = french_revolution.11
			100 = french_revolution.12
			100 = french_revolution.13
			100 = french_revolution.14
			100 = french_revolution.15
			100 = french_revolution.16
		}
	}
}
